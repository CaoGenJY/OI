思路（精炼）
把荷花看作完全图 (K_P) 的顶点，青蛙每次必须跳到不同的顶点（无自环），这是该图的邻接矩阵 (A)（对角线为0，非对角线为1）。
令 (a) 为从固定起点经过 (n) 次跳回起点的路径数，则对称性下所有起点相同，且
[
\operatorname{tr}(A^n)=P\cdot a.
]
利用矩阵 (A=J-I)（(J) 全1矩阵），其特征值为 (P-1)（重数1）和 (-1)（重数 (P-1)），所以
[
\operatorname{tr}(A^n)=(P-1)^n+(P-1)(-1)^n.
]
因此答案为
[
a=\frac{(P-1)^n+(P-1)(-1)^n}{P}.
]
对 (10^9+7) 取模时，使用快速幂计算 ((P-1)^n)，并用模逆（Fermat）除以 (P)。

复杂度： (O(\log n))。

下面给出 C++ 实现（处理大 (n)）：

```cpp
#include <bits/stdc++.h>
using namespace std;
using int64 = long long;
const int64 MOD = 1000000007;

int64 modpow(int64 a, long long e){
    a %= MOD;
    int64 r = 1;
    while(e){
        if(e & 1) r = (r * a) % MOD;
        a = (a * a) % MOD;
        e >>= 1;
    }
    return r;
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    long long P, n;
    if(!(cin >> P >> n)) return 0;
    int64 pm1 = (P - 1) % MOD;
    int64 t1 = modpow(pm1, n); // (P-1)^n mod MOD
    int64 neg1pow = (n % 2 == 0) ? 1 : (MOD - 1); // (-1)^n mod MOD
    int64 term = ( (t1 + (pm1 * neg1pow) % MOD) % MOD + MOD ) % MOD;
    // divide by P modulo MOD
    int64 invP = modpow(P % MOD, MOD - 2);
    int64 ans = (term * invP) % MOD;
    cout << ans << '\n';
    return 0;
}
```

示例检验：
输入 `10 2` 输出 `9`；输入 `10 3` 输出 `72`（与题目样例一致）。
